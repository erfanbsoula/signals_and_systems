import numpy as np
from matplotlib import pyplot as plt
from numpy import pi #to have it all along the program

Part a¶

In [2]:
k = 0.125
T = 1/k
# sampling frequency:
sf = 1
n = np.arange(0, 50, 1/sf)
start = int(T * sf)
stop = n.size - start
indxs = range(start, stop)
h_axis = n[start:stop] - T
In [3]:
x = np.cos(2*k*pi*n)
plt.stem(h_axis, x[start:stop])
plt.xlabel("n")
plt.title("x[n]")
#plt.savefig("p1_x1.jpg", dpi=360)
plt.show()
