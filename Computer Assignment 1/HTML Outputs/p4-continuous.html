sampling_steps = 0.2
sf = int(1/sampling_steps)
start = -10
stop = 10
t = np.arange(start, stop, sampling_steps)

def impulse_train(T):
    n = T*sf
    it = np.zeros((stop-start)*sf)
    for i in range(int(it.size/n)):
        it[i*n:(i+1)*n] = signal.unit_impulse(n, 'mid')
    return it
